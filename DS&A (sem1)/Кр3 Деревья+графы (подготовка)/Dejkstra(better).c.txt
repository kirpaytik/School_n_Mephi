#include <stdio.h>

int main()
{
    int n = 4;
    int matr[4][4] = {{0,10, 5, 0},
                      {0, 0,-6, 4},
                      {0, 0, 0,-7},
                      {0, 0, 0, 0}};
    int dist[n];
    int color[n];
    int prev[n];
    int i;
    for (i = 0; i < n; i++) {
        dist[i] = 99;
        color[i] = 0;
        prev[i] = 0;
    }
    dist[0] = 0;
    int from = 0, min;
    int count = 0;
    int flag = 1;
    while (flag) {
        flag = 0;
        for (count = 0; count < n; count++) {
            for (i = 0; i < n; i++) {
                if (!matr[from][i]) {
                    continue;
                }
                if (dist[i] > dist[from] + matr[from][i]) {
                    dist[i] = dist[from] + matr[from][i];
                    prev[i] = from + 1;
                    flag = 1;
                }
            }
            color[from] = 1;
            min = 100;
            for (i = 0; i < n; i++) {
                if (!color[i] && dist[i] < min) {
                    min = dist[i];
                    from = i;
                }
            }
        }
        for (i = 0; i < n; i++) {
            color[i] = 0;
            from = 0;
        }
    }
    printf("Вывод\ndist\n");
    for (i = 0; i < n; i++) {
        printf("%d\t", dist[i]);
    }
    printf("\ntrack\n");
    for (i = 0; i < n; i++) {
        printf("%d\t", prev[i]);
    }
    printf("\n");

    return 0;
}
